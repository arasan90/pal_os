44int pal_queue_create(pal_queue_t **queue, size_t item_size, size_t max_items)
45{
46 int ret_code = -1;
47 if (NULL != queue && 0 != item_size && 0 != max_items)
48 {
49 *queue = malloc(sizeof(struct pal_queue_s));
50 if (NULL != *queue)
51 {
52 (*queue)->data = malloc(item_size * max_items);
53 if (NULL != (*queue)->data)
54 {
55 (*queue)->item_size = item_size;
56 (*queue)->max_items = max_items;
57 (*queue)->head = 0;
58 (*queue)->tail = 0;
59 pthread_mutexattr_t attr;
60 pthread_mutexattr_init(&attr);
61 pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
62 pthread_mutex_init(&(*queue)->mutex, &attr);
63 pthread_cond_init(&(*queue)->full, NULL);
64 pthread_cond_init(&(*queue)->empty, NULL);
65 ret_code = 0;
66 }
67 else
68 {
69 free(*queue);
70 *queue = NULL;
71 }
72 }
73 }
74 return ret_code;
75}

77int pal_queue_enqueue(pal_queue_t *queue, void *const item, size_t timeout_ms, int from_isr)
78{
79 int ret_code = -1;
80 int error = 0;
81 (void)from_isr;
82 if (NULL != queue && NULL != item)
83 {
84 pthread_mutex_lock(&queue->mutex);
85 size_t free_slots = queue->max_items - (queue->tail - queue->head);
86 if (0 == free_slots)
87 {
88 if (PAL_OS_NO_TIMEOUT == timeout_ms)
89 {
90 error = 1;
91 }
92 else if (PAL_OS_INFINITE_TIMEOUT == timeout_ms)
93 {
94 pthread_cond_wait(&queue->full, &queue->mutex);
95 }
96 else
97 {
98 struct timespec timeout = {0};
99 clock_gettime(CLOCK_REALTIME, &timeout);
100 timeout.tv_sec += timeout_ms / 1000;
101 timeout.tv_nsec += (timeout_ms % 1000) * 1000000;
102 timeout.tv_sec += timeout.tv_nsec / 1000000000;
103 timeout.tv_nsec = timeout.tv_nsec % 1000000000;
104 if (ETIMEDOUT == pthread_cond_timedwait(&queue->full, &queue->mutex, &timeout))
105 {
106 error = 1;
107 }
108 }
109 }
110 if (!error)
111 {
112 int idx = queue->tail % queue->max_items;
113 memcpy((char *)queue->data + (idx * queue->item_size), item, queue->item_size);
114 queue->tail++;
115 ret_code = 0;
116 }
117 pthread_cond_signal(&queue->empty);
118 pthread_mutex_unlock(&queue->mutex);
119 }
120 return ret_code;
121}
