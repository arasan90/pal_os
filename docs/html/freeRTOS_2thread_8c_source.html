42UBaseType_t pal_thread_convert_priority(pal_thread_priority_t pal_priority)
43{
44 UBaseType_t freertos_priority = 1;
45 switch (pal_priority)
46 {
47 case PAL_THREAD_PRIORITY_LOW:
48 freertos_priority = configMAX_PRIORITIES / 3;
49 break;
50 case PAL_THREAD_PRIORITY_NORMAL:
51 freertos_priority = configMAX_PRIORITIES / 2;
52 break;
53 case PAL_THREAD_PRIORITY_HIGH:
54 freertos_priority = 2 * configMAX_PRIORITIES / 3;
55 break;
56 default:
57 break;
58 }
59 return freertos_priority;
60}

62void pal_thread_generic_func(void *const arg)
63{
64 pal_thread_t *thread = (pal_thread_t *)arg;
65 if (thread->func)
66 {
67 thread->state = PAL_THREAD_STATE_RUNNING;
68 thread->func(thread->arg);
69 }
70 xEventGroupSetBits(thread->event_group_handle, PAL_THREAD_THREAD_EXITED_EVENT_BITS);
71 vTaskDelete(NULL);
72}

74int pal_thread_create(pal_thread_t **thread, pal_thread_priority_t priority, size_t stack_size, pal_thread_func_t func, const char *name, void *arg)
75{
76 int ret_code = -1;
77 if (thread)
78 {
79 *thread = pvPortMalloc(sizeof(pal_thread_t));
80 if (*thread)
81 {
82 memset(*thread, 0, sizeof(pal_thread_t));
83 (*thread)->event_group_handle = xEventGroupCreate();
84 if ((*thread)->event_group_handle)
85 {
86 UBaseType_t freertos_priority = pal_thread_convert_priority(priority);
87 (*thread)->func = func;
88 (*thread)->arg = arg;
89 (*thread)->state = PAL_THREAD_STATE_STOPPED;
90 if (pdPASS == xTaskCreate(pal_thread_generic_func, name, stack_size / sizeof(StackType_t), arg, freertos_priority,
91 (TaskHandle_t *)(*thread)->thread_handle))
92 {
93 ret_code = 0;
94 }
95 else
96 {
97 vEventGroupDelete((*thread)->event_group_handle);
98 vPortFree(*thread);
99 *thread = NULL;
100 }
101 }
102 else
103 {
104 vPortFree(*thread);
105 *thread = NULL;
106 }
107 }
108 else
109 {
110 return -1;
111 }
112 }
113 return ret_code;
114}
